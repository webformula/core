  ${`import { Component } from '@webformula/core';
  import html from './page.html';

  export default class extends Component {
    static pageTitle = 'Binding variables';
    static html = html;
    
    basicBind = this.signal('');
    
    constructor() {
      super();
    }

    updateValue() {
      // Components will automatically listen for signals and render, so you do not need effects
      this.basicBind.value = 'Updated';
    }
  }`}
        
      
        
  ${`
    
  // Use signals and effects outside of the Component class
  
  import { Signal, effect } from '@webformula/core';


  const something = new Signal('some value');

  effect(() => {
    console.log(something.value);
  });

  something.value = 'new value';

  something.destroy();`}
